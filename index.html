Semantify.it Demo

This is a demo website for the semantify.it service. It shows how semantic annotations can be loaded at runtime and be crawled by engines like google's Data Testing Tool or semantify.it's semantic validator.

Sourcecode

Inject Annotation with basic Javascript

function appendAnnotation() {
    var element = document.createElement('script');
    element.type = 'application/ld+json';
    element.text = this.responseText;
    document.querySelector('head').appendChild(element);
}
var request = new XMLHttpRequest();
request.onload = appendAnnotation;
//here comes the short URL of your annotation
request.open("get", "https://smtfy.it/r1esOtnkhZ", true);
request.send();

Inject Annotation with JQuery

//here comes the short URL of your annotation
$.getJSON('https://smtfy.it/r1esOtnkhZ', function (annotation) {
    var element = document.createElement('script');
    element.type = 'application/ld+json';
    element.text = JSON.stringify(annotation);
    document.querySelector('head').appendChild(element);
});
